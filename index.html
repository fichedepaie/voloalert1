// ⚙️ CONFIGURAZIONE — modifica questi valori:
const TELEGRAM_TOKEN  = "IL_TUO_TOKEN_QUI";
const TELEGRAM_CHAT_ID = "IL_TUO_CHAT_ID_QUI";

// Parole chiave per riconoscere email Google Flights
const GF_KEYWORDS = [

// ═══════════════════════════════════════════════
// FUNZIONE PRINCIPALE — gira ogni 15 minuti
// ═══════════════════════════════════════════════
function checkFlightEmails() {
  const label = getOrCreateLabel("VoLoAlert/Processate");

  // Cerca email non lette da Google Flights
  const threads = GmailApp.search(
    'from:(noreply@google.com OR flights-noreply@google.com) ' +
    'subject:(price OR prezzo OR flight OR volo) ' +
    'is:unread',
    0, 20
  );

  if (threads.length === 0) {
    console.log("Nessuna email Google Flights da processare.");
    return;
  }

  console.log(`Trovate ${threads.length} email da processare.`);

  threads.forEach(thread => {
    const messages = thread.getMessages();
    messages.forEach(msg => {
      if (!msg.isUnread()) return;

      const from    = msg.getFrom();
      const subject = msg.getSubject();
      const body    = msg.getPlainBody().substring(0, 800);
      const date    = msg.getDate().toLocaleString("it-IT");

      console.log(`Processing: ${subject}`);

      // Invia a Telegram
      const sent = sendToTelegram(
        `✈️ ALERT GOOGLE FLIGHTS\n\n` +
        `📧 Da: ${from}\n` +
        `📋 Oggetto: ${subject}\n` +
        `⏰ ${date}\n\n` +
        `📄 
${escapeHtml(body)}
\n\n` +
        `🔗 Apri Google Flights`
      );

      if (sent) {
        msg.markRead();
        thread.addLabel(label);
        console.log(`✅ Inoltrato a Telegram: ${subject}`);
      }
    });
  });
}

// ═══════════════════════════════════════════════
// INVIA MESSAGGIO TELEGRAM
// ═══════════════════════════════════════════════
function sendToTelegram(text) {
  const url = `https://api.telegram.org/bot${TELEGRAM_TOKEN}/sendMessage`;
  const payload = {
    method: "post",
    contentType: "application/json",
    payload: JSON.stringify({
      chat_id: TELEGRAM_CHAT_ID,
      text: text,
      parse_mode: "HTML",
      disable_web_page_preview: false
    }),
    muteHttpExceptions: true
  };

  try {
    const response = UrlFetchApp.fetch(url, payload);
    const result = JSON.parse(response.getContentText());
    if (!result.ok) {
      console.error("Errore Telegram:", result.description);
      return false;
    }
    return true;
  } catch (e) {
    console.error("Errore rete:", e.toString());
    return false;
  }
}

// ═══════════════════════════════════════════════
// HELPER: crea/recupera label Gmail
// ═══════════════════════════════════════════════
function getOrCreateLabel(name) {
  let label = GmailApp.getUserLabelByName(name);
  if (!label) label = GmailApp.createLabel(name);
  return label;
}

function escapeHtml(text) {
  return text
    .replace(/&/g, "&")
    .replace(//g, ">");
}

// ═══════════════════════════════════════════════

  );
  if (ok) {
    console.log("✅ Test riuscito! Controlla Telegram.");
  } else {
    console.error("❌ Test fallito. Controlla Token e Chat ID.");
  }
}

🚀 Bonus: GitHub Actions (SerpApi 24/7)

Per usare SerpApi anche con il telefono spento, esporta gli alert e usa GitHub Actions.

1

Esporta alerts.json

Vai su scheda 🔔 Alert → "Esporta alerts.json"

2

Crea repository GitHub privato

github.com → New repository → nome: voloalert → Private → carica alerts.json

3

Aggiungi Secrets

Settings → Secrets → Actions → aggiungi: TELEGRAM_TOKEN, TELEGRAM_CHAT_ID, SERPAPI_KEY

4

Crea workflow

Crea file .github/workflows/monitor.yml con il pulsante qui sotto.

name: VoLoAlert Monitor

on:
  schedule:
    - cron: '0 */6 * * *'  # Ogni 6 ore
  workflow_dispatch:         # Esecuzione manuale

jobs:
  check-prices:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3

      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: '18'

      - name: Install dependencies
        run: npm install node-fetch

      - name: Check flight prices
        env:
          TELEGRAM_TOKEN: ${{ secrets.TELEGRAM_TOKEN }}
          TELEGRAM_CHAT_ID: ${{ secrets.TELEGRAM_CHAT_ID }}
          SERPAPI_KEY: ${{ secrets.SERPAPI_KEY }}
        run: node check-prices.js

Crea anche check-prices.js nella root del repo — esporta gli alert e usa lo stesso codice SerpApi dell'app.
